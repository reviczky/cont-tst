\startluacode

local list = {
    { entry = "AA"  },
    { entry = "Aa"  },
    { entry = "aa"  },
    { entry = "AAA" },
    { entry = "aaa" },
    { entry = "AaB" },
    { entry = "Aab" },
    { entry = "aab" },
    { entry = "AB"  },
    { entry = "Ab"  },
    { entry = "ab"  },
    { entry = "AbB" },
    { entry = "Abb" },
    { entry = "abb" },
}

sorters.setlanguage("en","zc,pc,zm,pm,uc")

for i=1,#list do
    list[i].split = { sorters.splitters.utf(list[i].entry) }
end

sorters.sort(list,sorters.comparers.basic)

context.blank()
for i=1,#list do
    context(list[i].entry)
    context.crlf()
end
context.blank()

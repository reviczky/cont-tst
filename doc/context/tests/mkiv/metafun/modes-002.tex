    draw unitsquare scaled u ;
    drawdblarrow (0,-0.2u)..(u,-0.2u) ;
    drawdblarrow (1.2u, 0)..(1.2u, u) ;

        label("e", (0.5u,-0.4u)) ;
        label("n", (1.6u, 0.5u)) ;

        label("n", (0.5u,-0.4u)) ;
        label("l", (1.6u, 0.5u)) ;
    fi ;
\stopuseMPgraphic

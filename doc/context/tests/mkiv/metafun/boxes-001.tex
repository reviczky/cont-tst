\starttext

\startMPpage
    input "boxes" ;
 %  loadmodule "xbox" ;

    path highlight ;

    boxit.A("Box A");
    boxit.B("Box B");
    boxit.C("Box C");

    defaultdx := 0pt;
    defaultdy := 0pt;

    A.c = origin;

    B.w - A.e = (1cm, 0);

    0.5[ A.s, B.s ] - C.n = (0, 1cm);

    draw image (drawunboxed(A,B,C));

    drawarrow A.s +- C.w;
    drawarrow B.s +- C.e;

    highlight := (A.sw -- B.se -- B.ne -- A.nw) enlarged EmWidth;

    draw highlight withcolor "darkgray";

\stopMPpage

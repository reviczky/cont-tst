\startluacode
    local p_strip = lpeg.stripper(lpeg.US("¹²³⁴⁵⁶⁷⁸⁹⁰"))

    interfaces.implement {
        name      = "StripNumber",
        public    = true,
        arguments = "string",
        actions   = function(str)
            context(lpeg.match(p_strip, str))
        end
    }
\stopluacode

\startformula
\int_{\Omega}f(x)dx = \int_{0}^\infty g(t)dt
\stopformula

If $\Omega:=(0,\infty)\times(0,\infty)$ then

\startformula
\iint_{\Omega}f(x)dx := \int_{0}^\infty \int_{0}^\infty g(s,t)dsdt
\stopformula

If $\Omega:=(0,\infty)\times(0,\infty)\times (0,\infty)$ then

\startformula
\iiint_{\Omega}f(x)dx= \int_{0}^\infty\int_{0}^\infty\int_{0}^\infty g(s,t,u)dsdtdu
\stopformula

\startformula
{\rm Res}(f,a) := {1 \over 2{\rm i}\pi }\oint_{\Gamma} {f(z) \over z-a}dz
\stopformula

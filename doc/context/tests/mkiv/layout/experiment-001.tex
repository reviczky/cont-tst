% Unfortunate we have no pdf viewers on tablets that:
%
% - default to fit width
% - remember where we were at a page when we swipe left/right
%
% so it's to soon to start thinking different about what a page
% is.

\definepapersize
  [onepage]
  [height=16000pt]

\definepapersize
  [thepage]
  [height=\the\dimexpr\pagetotal+\topspace+\bottomspace+1cm]

\setuppapersize
  [onepage]

\setuplayout
  [header=2cm,
   footer=2cm,
   topspace=2cm,
   bottomspace=2cm,
   width=middle,
   height=middle]

\appendtoks
    \setuppapersize
      [thepage]
\to \beforeeverypage

\useMPlibrary[dum]

      [onepage]

            \input ward  \par

            \input knuth \par
        }
        \page
    \stopchapter
}

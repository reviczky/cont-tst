local rehash = {
    ["right"] = "ok",
}

function xml.finalizers.tex.Shorter(collected,what)
    for i=1,#collected do
        if what == "all" then
            local str = xml.text(collected[i])
            context(rehash[str] or str)
        else
            context(str)
        end
    end
end

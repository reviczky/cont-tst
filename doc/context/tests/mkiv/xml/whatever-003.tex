\startluacode
    function xml.functions.make_tabulate(e)
        -- 'e' is the first a in a sequence
        local p = e.__p__
        local dt = p.dt
        local el = { }
        for i=e.ni,#dt do
            local ei = dt[i]
            if type(ei) == "table" then
                if ei.tg == "a" then
                    el[#el+1] = ei
                else
                    break
                end
            end
        end
        local three = false
        for i=1,#el do
            if xml.filter(el[i],"/b") then
                three = true
                break
            end
        end
        if three then
            context.starttabulate { "||||" }
        else
            context.starttabulate { "|||" }
        end
        for i=1,#el do
            local e = el[i]
            context.NC()
            if three then
                xml.cprint(xml.first(e,"/b"))
                context.NC()
            end
            xml.cprint(xml.first(e,"/c"))
            context.NC()
            xml.cprint(xml.first(e,"/d"))
            context.NC()
            context.NR()
        end
        context.stoptabulate()
    end

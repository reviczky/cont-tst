\startluacode
    function xml.finalizers.filter(e)
        local hash = { }
        for i=1,#e do
            for c in xml.collected(e[i],"/year") do
                local y = tonumber(c.at["value"])
                local h = hash[y]
                if not h then
                    h = { }
                    hash[y] = h
                end
                for c in xml.collected(c,"/album") do
                    local g = c.at.genre
                    local t = xml.text(c)
                    if h[g] then
                        h[g][t] = true
                    else
                        h[g] = { [t] = true }
                    end
                end
            end
        end
        -- ok, for utf you need a better sorter but for english this will do
        context.starttabulate { "|l|p|r|" }
        for k, v in table.sortedhash(hash) do
            context.HL()
            context.NC()
            context(k)
            context.NC()
            context.NC()
            context.NR()
            context.HL()
            for k, v in table.sortedhash(v) do
                local c = table.sortedkeys(v)
                context.NC()
                context(k)
                context.NC()
                context("%, t",c)
                context.NC()
                context(#c)
                context.NC()
                context.NR()
            end
            context.HL()
        end
        context.stoptabulate()
    end
\stopluacode

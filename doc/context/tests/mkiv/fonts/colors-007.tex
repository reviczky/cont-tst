\definefontfeature [MyFeaturesA] [default] [goodies={color-latin},colorscheme=default]
\definefontfeature [MyFeaturesB] [default] [goodies={color-latin},colorscheme=upper]
\definefontfeature [MyFeaturesC] [default] [goodies={color-latin},colorscheme=lower]

\definefont[MyFontA][SerifBold*MyFeaturesA]
\definefont[MyFontB][SerifBold*MyFeaturesB]
\definefont[MyFontC][SerifBold*MyFeaturesC]

\setfontcolorscheme[4]

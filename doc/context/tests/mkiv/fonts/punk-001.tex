\startluacode
    local byte, char, rep = string.byte, string.char, string.rep
    local function showtime(f,t)
        f = type(f) == "string" and byte(f) or f
        t = type(t) == "string" and byte(t) or t or f
        for i=f,t do
            context.par()
            context(rep(char(i),32))
            context.par()
        end
    end
    showtime("A","Z")
    showtime("a","z")
    showtime("0","9")
    showtime("!")
    showtime("?")
    showtime(".")
    showtime(",")
    showtime(":")
    showtime(";")
\stopluacode

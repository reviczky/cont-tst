\startluacode
    fonts.handlers.otf.addfeature {
        name    = "itest",
        type    = "chainsubstitution",
        prepend = 1,
        lookups = {
            {
                type = "multiple",
                data = {
                    ["f"] = { "f", 0x200C },
                },
            },
            {
                type = "substitution",
                data = {
                    ["a"] = "1",
                    ["b"] = "2",
                    ["c"] = "3",
                },
            },
            {
                type = "multiple",
                data = {
                    ["z"] = { "Z", "Z", "Z" },
                },
            },
            {
                type = "multiple",
                data = {
                    ["f"] = { "F", "F", "F" },
                },
            },
        },
        data = {
            rules = {
                {
                    current = { { "z" }, { "z" }, { "z" },  },
                    lookups = { 3 },
                },
                {
                    current = { { "f" }, { "f" }, { "f" },  },
                    lookups = { 3 },
                },
                {
                    current = { { "f" }, { "f" } },
                    lookups = { 1 },
                },
                {
                    current = { { "a" }, { "b" }, { "c" } },
                    lookups = { 2, false, 2 },
                },
            },
        }
    }
\stopluacode

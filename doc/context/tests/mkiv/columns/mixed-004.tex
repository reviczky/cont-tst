\definemarking[country]

\startsetups myheader
    (\getmarking[country][1][top]\enspace\emdash\enspace\getmarking[country][1][bottom])
    \hfill
    (\getmarking[country][2][top]\enspace\emdash\enspace\getmarking[country][2][bottom])
\stopsetups

\setupheadertexts
  [\setups{myheader}][]
  [][\setups{myheader}]

  {\par

   \par}
